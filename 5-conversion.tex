\question

\begin{parts}

\part[2]
Convert to radians in terms of $\pi$.

\begin{subparts}

\subpart
$\ang{150}$.

\subpart
$\ang{225}$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[1.5in]
A full rotation is $2\pi$ radians or $\ang{360}$, so the conversion is $2\pi = \ang{360}$. Now,
\[
	\frac{150}{360} = \frac{5}{12}, \qquad \frac{225}{360} = \frac{5}{8}.
\]
So,
\[
	\ang{150} = \frac{5}{12} \times 2\pi = \frac{5\pi}{6}, \qquad \ang{225} = \frac{5}{8} \times 2\pi = \frac{5\pi}{4}.
\]
\end{solutionorgrid}
\end{EnvFullwidth}

\end{subparts}

\part[2]
Convert to degrees.

\begin{subparts}

\subpart
$\displaystyle{\frac{11\pi}{8}}$.

\subpart
$\displaystyle{\frac{7\pi}{6}}$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[1.5in]
Note that $\pi = \ang{180}$, so
\[
	\frac{11\pi}{8} = \frac{11 \times \ang{180}}{8} = \ang{247.5}, \qquad \frac{7\pi}{6} = \frac{7 \times \ang{180}}{6} = \ang{210}.
\]
\end{solutionorgrid}
\end{EnvFullwidth}

\end{subparts}

\end{parts}
